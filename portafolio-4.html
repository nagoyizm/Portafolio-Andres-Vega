  function login() {
    let ingresoCliente = prompt("ingrese su nombre de usuario");
        let usuario_encontrado = false;
    
    for (let i = 0; i < cliente.length; i++) {
      if (ingresoCliente === cliente[i].nombre) {
        let ingresoClave = prompt("ingrese su clave");
        if (cliente[i].clave == ingresoClave) {
          usuario_encontrado = true;
        } else {
          alert("clave incorrecta");
          login();
        }
      } else {
        usuario_encontrado = false;
      }
      if (usuario_encontrado == true) {
          alert("Bienvenido a banca en linea");
          menuPrincipal();
          return
      } 
  
      function menuPrincipal() {
          let nuevoMenu = prompt(
            "opciones \n \n1. Ver Saldo \n 2.Realizar giro \n 3.Realizar deposito \n 4.Salir"
          );
          if (nuevoMenu == 1) {
            alert("su saldo es de " + cliente[i].saldo);
            menuPrincipal();
          } else if (nuevoMenu == 2) {
            let giro = prompt(
              "su saldo actual es de " +
                cliente[i].saldo +
                "\nIngrese el monto que desea retirar"
            );
            if (giro <= cliente[i].saldo) {
              cliente[i].saldo -= giro;
              alert("su nuevo saldo es de " + cliente[i].saldo);
              menuPrincipal();
            } else {
              alert("saldo insuficiente");
              menuPrincipal();
            }
          } else if (nuevoMenu == 3) {
            let deposito = prompt(
              "su saldo actual es de " +
                cliente[i].saldo +
                "\ncuanto desea depositar"
            );
            cliente[i].saldo += parseInt(deposito);
            alert("su nuevo saldo es de " + cliente[i].saldo);
            menuPrincipal();
          } else if (nuevoMenu == 4) {
            alert("gracias por usar nuestros servicios");
          } 
        }
    }
  }
