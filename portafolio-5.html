//funcion para listar las tareas dentro de una tabla, lo que hace es escribirla
function tabla(tareas) {
  let cuerpoTabla = document.getElementById("cuerpo-tabla");
  cuerpoTabla.innerHTML = "";
  //tareas es el array y tarea es el objeto
  for (let i = 0; i < tareas.length; i++) {
    cuerpoTabla.innerHTML +=`<tr>
    <td >${tareas[i].tarea}</td>
    <td> <button class="btn btn-danger border px-3" onclick="eliminarTarea({})"">Finalizar</button>
                            `;
                            
  }
}

//invocar la funcion
tabla(tareas);

//funcion para agregar una tarea
function abrirFormulario() {
  let formulario = document.getElementById("formulario");
  if (formulario.style.display == "none") {
    formulario.style.display = "block";
  } else {
    formulario.style.display = "none";
  }
  //if ternario
  //formulario.style.display=(formulario.style.display=="none")?"block":"none";
}

function agregarTarea() {
  let nuevaTarea = document.getElementById("nuevaTarea");
  console.log(nuevaTarea);
  tareas.push({ tarea: nuevaTarea.value });
  tabla(tareas);
  nuevaTarea.value = "";
  abrirFormulario();
}

function eliminarTarea(indice){
  let confirmar=confirm("¿Estás seguro de que quieres eliminar esta tarea?");
  //devuleve un valor booleando
  if(confirmar === true){
    tareas.splice(indice,1);
    tabla(tareas);}

}
